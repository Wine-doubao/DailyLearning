## 考点：闭包

1. ```javascript
   function Foo(){
     var i=0;
     return function(){
       console.log(i++);
     }
   }
   var f1=Foo(),f2=Foo();
   f1();//0
   f1();//1
   f2();//0
   ```



## 考点：setTimeout，Promise

1. ```javascript
   setTimeout(()=>{
     console.log(1);
   })
   const p1 = Promise.resolve(()=>{
     console.log(2);
   })
   const p2 = new Promise((resolve,reject)=>{
     console.log(3);
     resolve();
   })
   Promise.race([p1,p2]).then(()=>{
     console.log(4);
   })
   Promise.all([p1,p2]).then(()=>{
     console.log(5);
   })
   console.log(6);
   //3 6 4 5 1
   ```
